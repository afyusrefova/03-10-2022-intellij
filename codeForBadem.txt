public class Badem {

    int size;
    String type;

    public Badem(int size, String type){
        this.size = size;
        this.type = type;
    }

    public void setSize(int size) {
        this.size = size;
    }

    public void setType(String type) {
        this.type = type;
    }

    public int getSize() {
        return size;
    }

    public String getType() {
        return type;
    }
}
